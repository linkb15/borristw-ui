## Installation

<Tabs>
  <TabItem label="CLI">
    <Steps>
      1.  Run the following command:

          <PackageManagerTabs>
            <Fragment slot="npm">
            ```bash frame="none"
            npx borristw-ui@latest add accordion
            ```
            </Fragment>
            <Fragment slot="pnpm">
            ```bash frame="none"
            pnpm dlx borristw-ui@latest add accordion
            ```
            </Fragment>
            <Fragment slot="yarn">
            ```bash frame="none"
            npx borristw-ui@latest add accordion
            ```
            </Fragment>
          </PackageManagerTabs>

      2.  Update `tailwind.config.js`

          Add the following animations to your `tailwind.config.js` file:

          ```diff lang="js" title="tailwind.config.js" ins={5-18}
          /** @type {import('tailwindcss').Config} */
          module.exports = {
            theme: {
              extend: {
                keyframes: {
                  'accordion-down': {
                    from: { height: '0' },
                    to: { height: 'var(--height)' },
                  },
                  'accordion-up': {
                    from: { height: 'var(--height)' },
                    to: { height: '0' },
                  },
                },
                animation: {
                  'accordion-down': 'accordion-down 0.2s ease-out',
                  'accordion-up': 'accordion-up 0.2s ease-out',
                },
              },
            },
          };
          ```
    </Steps>

  </TabItem>

## Usage

```tsx
import {
  Accordion,
  AccordionContent,
  AccordionItem,
  AccordionTrigger,
} from '@/components/ui/accordion';
```

```tsx
<Accordion type="single" collapsible>
  <AccordionItem value="item-1">
    <AccordionTrigger>Is it accessible?</AccordionTrigger>
    <AccordionContent>
      Yes. It adheres to the WAI-ARIA design pattern.
    </AccordionContent>
  </AccordionItem>
</Accordion>
```
